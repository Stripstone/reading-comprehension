// ===================================
// ⚡ TTS Speed Select
// ===================================
(function initSpeedSelect() {
  const select = document.getElementById('shell-speed-select');
  if (!select) return;

  // Restore persisted speed on boot.
  try {
    const saved = parseFloat(localStorage.getItem('rc_tts_speed') || '');
    if (saved > 0 && isFinite(saved)) {
      TTS_STATE.playbackRate = saved;
      // Select the closest matching <option>
      const opt = Array.from(select.options).find(o => parseFloat(o.value) === saved);
      if (opt) opt.selected = true;
    }
  } catch (_) {}

  select.addEventListener('change', () => {
    const rate = parseFloat(select.value) || 1;
    TTS_STATE.playbackRate = rate;
    // Apply immediately if cloud audio is currently playing.
    try { if (TTS_STATE.audio) TTS_STATE.audio.playbackRate = rate; } catch (_) {}
    try { localStorage.setItem('rc_tts_speed', String(rate)); } catch (_) {}
  });
})();

// Called by onclick="handlePausePlay()" on #shell-pause-btn.
// Delegates to ttsPause() / ttsResume() and keeps the button label in sync.
window.handlePausePlay = function handlePausePlay() {
  if (!TTS_STATE || !TTS_STATE.activeKey) return;
  const btn = document.getElementById('shell-pause-btn');

  const cloudPlaying  = !!(TTS_STATE.audio && !TTS_STATE.audio.paused);
  const browserPlaying = !!(typeof speechSynthesis !== 'undefined' &&
                             speechSynthesis.speaking && !speechSynthesis.paused);
  const isPlaying = cloudPlaying || browserPlaying;

  if (isPlaying) {
    ttsPause();
    _syncPauseLabel(btn, false);
  } else {
    ttsResume();
    _syncPauseLabel(btn, true);
  }
};

function _syncPauseLabel(btn, playing) {
  if (!btn) return;
  btn.title = playing ? 'Pause narration' : 'Resume narration';
  // Walk childNodes backwards to find the text node (first is SVG, second is text).
  const nodes = btn.childNodes;
  for (let i = nodes.length - 1; i >= 0; i--) {
    if (nodes[i].nodeType === 3 && nodes[i].textContent.trim()) {
      nodes[i].textContent = playing ? ' Pause' : ' Resume';
      return;
    }
  }
}

window.handleAutoplayToggle = function handleAutoplayToggle() {
  const checkbox = document.getElementById('autoplayToggle');
  if (!checkbox) return;
  checkbox.checked = !checkbox.checked;
  checkbox.dispatchEvent(new Event('change'));
  const btn = document.getElementById('shell-autoplay-btn');
  if (btn) btn.classList.toggle('pb-active', AUTOPLAY_STATE.enabled);
};

Why it is risky

This code is shell-specific, not generic UI infrastructure.

Risks in this snippet
Hard-codes shell DOM IDs:
shell-speed-select
shell-pause-btn
shell-autoplay-btn
autoplayToggle
Assumes globals already exist:
TTS_STATE
AUTOPLAY_STATE
ttsPause()
ttsResume()
Boot-order sensitive:
immediately runs IIFE on load
can silently no-op depending on script order / DOM timing
Writes directly into persistence:
localStorage.rc_tts_speed
Mixes UI adapter logic with engine state logic
childNodes text-node mutation is fragile
breaks if button markup changes
Autoplay toggle is indirect
flips hidden checkbox and dispatches change
depends on another handler being present and correct
Uses TTS_STATE.playbackRate
if the rest of system uses a different property name elsewhere, this can drift
What is safe about it

It is trying to reuse existing system controls rather than invent a new playback engine:

pause/play delegates to ttsPause() / ttsResume()
autoplay delegates through the existing checkbox handler
speed applies to current audio and persisted preference

So it is not wildly off-architecture.
But it is still the wrong layer for monolith-first work.

If your rule is:

unified monolith first, shell adapters later

then this should be treated as:

“Functional but misplaced.”
